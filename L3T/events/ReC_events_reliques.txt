province_event = {
    id = 	REC.0101
	hide_window = yes
	
	trigger = {	
		NOT = {	
			has_global_flag = relique_trouvee_01
			has_province_flag = vraie_relique01
			has_province_modifier = relique01
			has_province_flag = a_deja_une_relique
			religion_group = zoroastrian_group
			any_neighbor_province = { has_province_flag = a_deja_une_relique }
			religion_group = pagan_group
			religion_group = indian_group
			religion_group = jewish_group
			}
		}
		
	mean_time_to_happen = {
		months = 6
		modifier = { factor = 15.50 religion_group = muslim }
		modifier = { factor = 1.20 religion_group = christian }
		modifier = { factor = 5.50 religion = orthodox }
		modifier = { factor = 0.50 religion = catholic }
		modifier = { factor = 3.50 has_province_flag = relique_testee }
		modifier = { # There is an holy site in the province (more exactly a holy site for the baron)
			factor = 0.50 
			any_province_holding = {
				owner = {
					PREV = {
						is_holy_site = PREV
					}
				}
			}
		}
	}
	
	immediate = {
		#name = "REC.0101A"
		#hidden_tooltip = {	
			random_list = {
			96 = { } #la relique n'est pas authentique, pas de notif pour limiter le spam
			 1 = { 
			 set_province_flag = fausse_relique 
			 set_province_flag = a_deja_une_relique
			 } #la relique n'est pas authentique, notification pour les catholiques uniquement
			 3 = { 
			 clr_province_flag = fausse_relique
			 set_province_flag = vraie_relique01 
			 set_province_flag = a_deja_une_relique
			 set_global_flag =	relique_trouvee_01	} #notification de relique authentique pour catholique et musulmans
			}
		set_province_flag = relique_testee #sert à renouveler le test plus tard
		#} 
		}
}
