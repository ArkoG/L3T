		random_demesne_title = {
			limit = {
				tier = baron
				is_republic = yes
				NOT = {
					has_building = CITE_L1
					has_building = CITE_L2
					has_building = VOC_INDUSTRIELLE
					has_building = VOC_COMMERCIALE
				}

				location = {
					NOT = {
						any_province_holding = {
							tier = baron
							is_republic = yes
							has_building = VOC_COMMERCIALE
						}
						any_neighbor_province = {
							any_province_holding = {
								tier = baron
								is_republic = yes
								has_building = VOC_COMMERCIALE
							}
						}
					}
				}
			}
			add_building = CITE_L1
			add_building = VOC_COMMERCIALE

			location = { add_province_modifier = { name = 20resourcesPLUS duration = -1 }}
		}
	}

#domaine Communal province à double cité  
province_event = {
    id = L3T.00041
	hide_window = yes
	
	trigger = {
		NOT = { has_province_flag = check_double_city }		
	}
	
	mean_time_to_happen = {	days = 100 }
	
	immediate = {
		set_province_flag = check_double_city
		# Compte le nombre de cités
		set_variable = { which = "cityNb" value = 0 }
		any_province_holding = {
			if = {
				limit = { 
					tier = baron
					holding_type = city
				} 
				ROOT = { change_variable = { which = "cityNb" value = 1 } }
			}
		}
		
		# S'il y a plus de 2 cités (supérieur ou égal)
		if = {
			limit = {
				check_variable = { which = "cityNb" value = 2 }
			}
			
			# Une cité sera commerçante
			random_province_holding = {
				limit = {
					tier = baron
					holding_type = city
				}
				add_building = CITE_L1
				add_building = VOC_COMMERCIALE
				# MOUCHARD # TODO blocA mais ATTENTION gestion du voisinage
				location = { add_province_modifier = { name = 17resources duration = -1 }}
			}
			
			# Les autres productrices
			any_province_holding = {
				if = {
					limit = {
						tier = baron
						holding_type = city
						NOT = { has_building = VOC_COMMERCIALE }
					}
					add_building = CITE_L1
					add_building = VOC_INDUSTRIELLE
				}
			}
		}
	}
}
