province_event = {
    id = REC.999
	#desc = "DESCREC.999"
	#picture = GFX_evt_L3T_city_visitor
	#border = GFX_event_normal_frame_religion
	hide_window = yes
	
	mean_time_to_happen = { days = 15 }
	
	trigger = {
		owner = { ai = no }
		OR = {
			NOT = { has_province_modifier = relique01 }
			NOT = { has_province_modifier = relique02 }
			NOT = { has_province_modifier = relique03 }
			NOT = { has_province_modifier = relique04 }
			NOT = { has_province_modifier = relique05 }
			NOT = { has_province_flag = relique }
			}
	}

	immediate = {
		#name = "REC.999A" #
		set_province_flag = relique
		add_province_modifier = { name = relique_generique duration = -1 }
		random_list = {
		#if = { limit = { NOT = { has_province_modifier = relique01 } }
		10 = {	add_province_modifier = { name = relique01 duration = -1 }
			owner = { character_event = { id = REC.1001 days = 1 }}
		}
		#if = { limit = { NOT = { has_province_modifier = relique02 } }
		10 = {	add_province_modifier = { name = relique02 duration = -1 }
			owner = { character_event = { id = REC.1001 days = 1 }}
		}
		#if = { limit = { NOT = { has_province_modifier = relique03 } }
		10 = {	add_province_modifier = { name = relique03 duration = -1 }
			owner = { character_event = { id = REC.1001 days = 1 }}
		}
		#if = { limit = { NOT = { has_province_modifier = relique04 } }
		10 = {	add_province_modifier = { name = relique04 duration = -1 }
			owner = { character_event = { id = REC.1001 days = 1 }}
		}
		#if = { limit = { NOT = { has_province_modifier = relique05 } }
		10 = {	add_province_modifier = { name = relique05 duration = -1 }
			owner = { character_event = { id = REC.1001 days = 1 }}
		}
		}
	}
}
