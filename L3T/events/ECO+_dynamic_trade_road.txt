namespace = TRADE

# Give the step A of the trade roads
province_event = {
    id = TRADE.0001
	hide_window = yes
	
	trigger = {
		NOT = { has_province_flag = trade_road_step_A }
		OR = {
			province_id = 112 # Ile de France
			province_id = 228 # Dauphine Viennois
		}		
	}
		
	mean_time_to_happen = {	days = 10 }	

	immediate = {
		set_province_flag = trade_road_step_A
		add_province_modifier = { name = route_terrestre duration = -1 }	
	}
}

# Give the step B of the trade roads
province_event = {
    id = TRADE.0002
	hide_window = yes
	
	trigger = {
		has_province_flag = trade_road_step_A
	}
		
	mean_time_to_happen = {	days = 10 }	

	immediate = {
		random_neighbor_province = {
			limit = {
				NOT = {
					has_province_modifier = route_terrestre
					has_province_flag = trade_road_step_B
					any_neighbor_province = { has_province_flag = trade_road_step_B }
				}
			}
			set_province_flag = trade_road_step_B
			add_province_modifier = { name = route_terrestre duration = -1 }
		}
	}
}

# Give the step C of the trade roads
# COPY-PASTE with adaptation of TRADE.0002
province_event = {
    id = TRADE.0003
	hide_window = yes
	
	trigger = {
		has_province_flag = trade_road_step_B
	}
		
	mean_time_to_happen = {	days = 10 }	

	immediate = {
		random_neighbor_province = {
			limit = {
				NOT = {
					# To not give step C to the province which gave it the step B
					has_province_modifier = route_terrestre
					
					# To not give step C to a province which have already it
					has_province_flag = trade_road_step_C
					
					# To do some branches and not circles
					any_neighbor_province = { has_province_flag = trade_road_step_C }
					
					# To not give step C to a province which don't have a minimal distance of 2 (in terms of neighboor) to a step A province
					any_neighbor_province = { has_province_flag = trade_road_step_A }
					any_neighbor_province = { has_province_flag = trade_road_step_D }
					any_neighbor_province = { has_province_flag = trade_road_step_E }
					any_neighbor_province = { has_province_flag = trade_road_step_F }
					any_neighbor_province = { has_province_flag = trade_road_step_G }
					any_neighbor_province = { has_province_flag = trade_road_step_H }
					any_neighbor_province = { has_province_flag = trade_road_step_I }
					any_neighbor_province = { has_province_flag = trade_road_step_J }					
				}
			}
			set_province_flag = trade_road_step_C
			add_province_modifier = { name = route_terrestre duration = -1 }
		}
	}
}
