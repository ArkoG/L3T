character_event = {
	id = L3T.230
	hide_window = yes
	is_triggered_only = yes
	
	trigger = {
		FROM = { 
			tier = baron	
			holding_type = castle 
			has_building = ca_tour_L4
			NOT = { has_title_flag = BONUS_ca_tour_L4 }
		}
	}
	
	immediate = {
		FROM = { 
			set_title_flag = BONUS_ca_tour_L4
			holder_scope = {
				prestige = -90
			}
		}
	}
}
