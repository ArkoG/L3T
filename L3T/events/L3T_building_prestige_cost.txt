#holding_type = castle 
character_event = {
	id = L3T.230
	hide_window = yes
	is_triggered_only = yes
	
	trigger = {
		FROM = { 
			tier = baron	
			holding_type = castle 
			has_building = ca_tour_L4
		}
		NOT = { FROM = { holder_scope = { has_character_flag = BONUS_ca_tour_L4 } } }
	}
	
	immediate = {
		FROM = { 
			holder_scope = { 
				set_character_flag = BONUS_ca_tour_L4
				prestige = -90
			}
		}
	}
}
