namespace = REC

# Artist startup
character_event ={
    id = REC.2100
	hide_window = yes
	
	trigger = {
		NOT = { had_global_flag = {flag = artist_init_generation days = 30 } }
	}

	is_triggered_only = yes	
	
	only_independent = yes
	
	immediate = {
		set_global_flag = artist_init_generation
		if = {
			limit = {
				religion = catholic
				independent = yes
			}
			
			# Create apprentis
			random_list = {
				76 = { }
				8 = {
					create_character = {
						religion = catholic
						random_traits = yes
						trait = apprenti_sculpteur
						age = 12
						dynasty = none
						female = no
					}
					new_character = {
						give_nickname = nick_le_sculpteur
						join_society = artist_guild
					}
				}
				8 = {
					create_character = {
						religion = catholic
						random_traits = yes
						trait = apprenti_verrier
						age = 12
						dynasty = none
						female = no
					}
					new_character = {
						give_nickname = nick_le_verrier
						join_society = artist_guild
					}
				}
				8 = {
					create_character = {
						religion = catholic
						random_traits = yes
						trait = apprenti_fresquiste
						age = 12
						dynasty = none
						female = no
					}
					new_character = {
						give_nickname = nick_le_fresquiste
						join_society = artist_guild
					}
				}
			}
			
			# Create compagnons
			random_list = {
				85 = { }
				5 = {
					create_character = {
						religion = catholic
						random_traits = yes
						trait = compagnon_sculpteur
						age = 30
						dynasty = none
						female = no
					}
				}
				5 = {
					new_character = {
						give_nickname = nick_le_sculpteur
						join_society = artist_guild
						society_rank_up = 1
					}
					create_character = {
						religion = catholic
						random_traits = yes
						trait = compagnon_verrier
						age = 30
						dynasty = none
						female = no
					}
					new_character = {
						give_nickname = nick_le_verrier
						join_society = artist_guild
						society_rank_up = 1
					}
				}
				5 = {
					create_character = {
						religion = catholic
						random_traits = yes
						trait = compagnon_fresquiste
						age = 30
						dynasty = none
						female = no
					}
					new_character = {
						give_nickname = nick_le_fresquiste
						join_society = artist_guild
						society_rank_up = 1
					}
				}
			}
			
			# Create maitres
			random_list = {
				91 = { }
				3 = {
					create_character = {
						religion = catholic
						random_traits = yes
						trait = maitre_sculpteur
						age = 50
						dynasty = none
						female = no
					}
					new_character = {
						give_nickname = nick_le_sculpteur
						join_society = artist_guild
						society_rank_up = 2
					}
				}
				3 = {
					create_character = {
						religion = catholic
						random_traits = yes
						trait = maitre_verrier
						age = 50
						dynasty = none
						female = no
					}
					new_character = {
						give_nickname = nick_le_verrier
						join_society = artist_guild
						society_rank_up = 2
					}
				}
				3 = {
					create_character = {
						religion = catholic
						random_traits = yes
						trait = maitre_fresquiste
						age = 50
						dynasty = none
						female = no
					}
					new_character = {
						give_nickname = nick_le_fresquiste
						join_society = artist_guild
						society_rank_up = 2
					}
				}
			}			
	
			if = {
				limit = { # Restricted to the Pope to be executed only once
					religion = catholic
					controls_religion = yes
				}
				# Remove non-artists of the guild
				any_society_member = {
					limit = {
						society_member_of = artist_guild
						NOT = {
							trait = apprenti_sculpteur
							trait = compagnon_sculpteur
							trait = maitre_sculpteur
							trait = apprenti_fresquiste
							trait = compagnon_fresquiste
							trait = maitre_fresquiste
							trait = apprenti_verrier
							trait = compagnon_verrier
							trait = maitre_verrier
						}
					}
					leave_society = yes
				}
			}
		}
	}
}

# Death of a master glass maker
# ROOT is the dead master glass maker
character_event ={
    id = REC.2104
	hide_window = yes

	is_triggered_only = yes	
	
	trigger = {
		trait = maitre_verrier
		has_character_modifier = work_on_cathedral 
	}
	
	immediate = {
		employer = {
			if = {
				limit = {
					any_courtier = {
						trait = maitre_verrier
						has_character_modifier = work_on_cathedral 
						NOT = { character = ROOT }
					}
				}
				character_event = { id = REC.2106 }
				break = yes
			}
			character_event = { id = REC.2105 }
		}
	}	
}

# Employer has no more master glass maker
character_event ={	
	id = REC.2105 
	desc = "DESCREC.2105"
	picture = GFX_evt_REC5

	is_triggered_only = yes
	
	option = {
		name = "REC.2105A"
		trigger = {
			any_courtier = {
				trait = compagnon_verrier
				has_character_modifier = work_on_cathedral 
			}
		}
		ai_chance = {
			factor = 40
			modifier = {
				factor = 0
				NOT = { wealth = 30 } 
			}
		}
		wealth = -30
		prestige = 10
		random_courtier = {
			limit = {
				trait = compagnon_verrier
				has_character_modifier = work_on_cathedral 
			}
			remove_trait = compagnon_verrier
			add_trait = maitre_verrier
			hidden_tooltip = { change_society_currency = -1250 }
		}
	}
	
	option = {
		name = "REC.2105B"
		trigger = { available_master_glass_maker = yes }
		ai_chance = {
			factor = 30
			modifier = {
				factor = 0
				NOT = { wealth = 50 } 
			}
		}
		hire_master_glass_maker = yes
	}
	
	option = {
		name = "REC.2105C"
		ai_chance = {
			factor = 30
			modifier = {
				factor = 2
				trait = greedy
			}
		}
		custom_tooltip = { text = TOOLTIPREC.2105C }
	}
}
