	desc = "DESCL3T.81"

				liege =  { character_event = { id = L3T.86 days = 10 }} #informe le lige du retour
				#noble is informed about the flee
				FROM = { character_event = { id = L3T.87 days = 5 } }
			}
			25 = {
				#failed and was catched
				cancel_job_action = action_fabricate_claims
				character_event = { id = L3T.82 days = 3 random = 2 } #informe de l'échec

				liege =  { character_event = { id = L3T.83 days = 5 random = 3 }} #informe le lige de l'arrestation et du retour
				#inform noble about success about successful arrestation
				FROM = { character_event = { id = L3T.84 days = 5 } }
			}
		}
	}
	option = {
		name = "L3T.81B"
		#no worries
		ai_chance = {
			factor = 35
		}
		#to hide
		random_list = {
			25 = {
				#success, false rumour ?
				character_event = { id = L3T.88 days = 15 random = 5 } #informe du succès, can get bolder
			}
			75 = {
				#was catched
				cancel_job_action = action_fabricate_claims
				character_event = { id = L3T.82 days = 5 random = 2 } #informe de l'échec
				#inform liege
				liege =  { character_event = { id = L3T.83 days = 10 random = 3 }} #informe le lige de l'arrestation et du retour
				#inform noble
				FROM = { character_event = { id = L3T.84 days = 7 } }
			}
		}
	}
}
