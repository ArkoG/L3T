# spécialisation Forêts -pour les Féodaux
province_event = {
    id = XYLO.2
	desc = "DESCXYLO.2"
	picture = GFX_evt_XYLO_forest2

	trigger = {
		has_province_modifier =	foret1
		NOT = {
			has_province_modifier = foret2_expl
			has_province_modifier = foret2_chasse
			has_province_modifier = foret2_holy				
			}
		
		owner = {
				is_feudal = yes
				is_republic = no
				#is_patrician = no
				is_theocracy = no				
				OR = {
					any_courtier = {
						has_minor_title = title_forestier
					}
					any_vassal = {
						has_minor_title = title_forestier
					}
				}
			}
	}
		
	mean_time_to_happen = {
		days = 300
	}	

option = {
		name = "XYLO.2A"	
		trigger = { region = world_europe }
		ai_chance = {
			factor = 70
				modifier = { factor = 0 NOT = { religion_group = christian }}
				modifier = { factor = 2 owner = { trait = hunter }}
		}		#à pondérer. choix privilégié si trait chasseur	et a nommé un maître de chasse.
		add_province_modifier = {
			name = foret2_chasse 
			duration = -1
			}
	}
option = {
		name = "XYLO.2B"	
		trigger = { region = world_europe }
		ai_chance = { 
			factor = 30
			modifier = { factor = 2 owner = { stewardship = 10 }}
			modifier = { factor = 2 owner = { stewardship = 15 }}
			}		#à pondérer. choix privilégié si bon niveau d'intendance
		add_province_modifier = {
			name = foret2_expl 
			duration = -1
			hidden = yes
			}
	}
option = {
		name = "XYLO.2C"	
		trigger = { NOT = { region = world_europe }}
		ai_chance = { factor = 30 }	
		add_province_modifier = {
			name = foret2_expl 
			duration = -1
			hidden = yes
			}
	}	
}
