character_event = { #notification for 1103 disease decrease
	id = L3T.11032
	title = "NAMEL3T.11032"

	desc = {
		trigger = { has_global_flag = debug_L3T }
		text = DESCL3T.11032_debug
	}
	desc = {
		trigger = {  
			NOT = { has_global_flag = debug_L3T }
			event_target:holding_dev = {
				location = {
					has_disease = yes
					NOT = {	has_province_modifier = meteo_inondation }
					NOT = {	has_province_modifier = meteo_secheresse }
				}
			}
		}
		text = DESCL3T.11032_disease
	}
	desc = {
		trigger = {  
			NOT = { has_global_flag = debug_L3T }
			event_target:holding_dev = {
				location = {
					has_disease = no
					OR = {
						has_province_modifier = meteo_inondation
						has_province_modifier = meteo_secheresse
					}
				}
			}
		}
		text = DESCL3T.11032_chatastrophe
	}
	desc = {
		trigger = {  
			NOT = { has_global_flag = debug_L3T }
			event_target:holding_dev = {
				location = {
					has_disease = yes
					OR = {
						has_province_modifier = meteo_inondation
						has_province_modifier = meteo_secheresse
					}
				}
			}
		}
		text = DESCL3T.11032_chatastrophe #in case of both disease+meteo, meteo desc
	}
	picture = GFX_evt_castle_construction
	border = GFX_event_normal_frame_economy
	
	is_triggered_only = yes
	
	trigger = {
		has_global_flag = init_population
	}
	option = {
		name = "L3T.11032a"
		clear_event_target = holding_dev 
	}
}
