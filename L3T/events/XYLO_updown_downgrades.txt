# downgrade -don de terres boisées à l'Eglise- v1
province_event = { 
    id = XYLO.0107
	desc = "DESCXYLO.0107"
	picture = GFX_evt_XYLO_forest1

	trigger = {
		has_province_modifier = foret_up
		NOT = { had_province_flag = { flag = transaction_forestiere_recente years = 5 }}
		owner = {
			NOT = { has_character_flag = don_refuse }
			 OR = { religion = catholic is_heresy_of = catholic }
			NOT = { piety = 50 }
		}
	}
		
	mean_time_to_happen = { months = 15 }	

	immediate = { set_province_flag = transaction_forestiere_recente }	
	
	option = {
		name = "XYLO.0107A"	 #non
		ai_chance = { factor = 60 }
		owner = { set_character_flag = don_refuse }
	}
	option = {
		name = "XYLO.0107B"	#oui
		ai_chance = { factor = 40 }
		remove_province_modifier = foret_up
		owner = { piety = 75 }
		religion_authority = { modifier = don_foret		years = 5 }
		}
	option = {
		name = "XYLO.0107C"	#vente
		ai_chance = { factor = 40 }
		trigger = { 
			owner = { 
				OR = { 
					trait = cynical
					trait = greedy
					trait = imbecile
					trait = slow
					}
				}
			}
		remove_province_modifier = foret_up
		owner = { piety = -50 }
		owner = { wealth = 150 }
		}
}
