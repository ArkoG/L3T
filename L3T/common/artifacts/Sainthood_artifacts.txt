###sainthood relics

ancestor_relic_0 = {
	monthly_character_prestige = 0.2
	monthly_character_piety = 0.3
	quality = 1
	active = {
		OR = { # Correct religion to become a Saint with an appropriate head
			religion = catholic
			AND = {
				religion = fraticelli
				is_title_active = d_fraticelli
			}
		}
		any_dynasty_member_even_if_dead  = { 
			dynasty = ROOT
			trait = saint
		}
	}
	allowed_gift = { always = no }
	flags = { dynastic_relic relique }
	stacking = no
	picture = GFX_relic_box_simple
	slot = slot_relique
}
